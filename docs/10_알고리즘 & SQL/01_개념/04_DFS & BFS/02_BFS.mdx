class Queue {
    constructor() {
      this.items = {};
      this.frontIndex = 0;
      this.backIndex = 0;
    }
  
    enqueue(item) {
      this.items[this.backIndex] = item;
      this.backIndex++;
    }
  
    dequeue() {
      if (this.isEmpty()) return undefined;
      
      const item = this.items[this.frontIndex];
      delete this.items[this.frontIndex];
      this.frontIndex++;
      return item;
    }
  
    isEmpty() {
      return this.frontIndex === this.backIndex;
    }
  
    size() {
      return this.backIndex - this.frontIndex;
    }
}

export function bfs(g, s, visit) {
    const q = new Queue();
    q.enqueue(s);
    
    while (!q.isEmpty()) {
      const c = q.dequeue(); // O(1) 연산
      result.push(c);
      
      for (const n of g[c]) {
        if (!visit[n]) {
            visit[n] = true;
            q.enqueue(n);
        }
      }
    }
}
